## Usage

Add Your Social Links to the `socialLinks` array in the `SocialShare` component.

```tsx showLineNumbers
import { SocialShare } from "@/components/ui/social-share";
```

```tsx showLineNumbers
import {
  IconBrandGithub,
  IconBrandLinkedin,
  IconBrandWhatsapp,
  IconBrandX,
} from "@tabler/icons-react";

const phoneNumber = process.env.NEXT_PUBLIC_PHONE_NUMBER;
const socialLinks = [
  {
    name: "Github",
    icon: <IconBrandGithub />,
    link: "https://github.com/yourusername",
    color: "bg-black",
  },
  {
    name: "LinkedIn",
    icon: <IconBrandLinkedin />,
    link: "https://linkedin.com/in/yourusername",
    color: "bg-blue-500",
  },
  {
    name: "WhatsApp",
    icon: <IconBrandWhatsapp />,
    link: `https://wa.me/${phoneNumber}`,
    color: "bg-green-500",
  },
  {
    name: "X",
    icon: <IconBrandX />,
    link: "https://x.com/yourusername",
    color: "bg-black",
  },
];

<SocialShare socialLinks={socialLinks} />;
```
