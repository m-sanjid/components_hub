## Usage

```tsx
import { Navbar } from "@/components/ui/navbar";
```

```tsx
const navItems = [
  { href: "/projects", label: "Projects" },
  { href: "/services", label: "Services" },
  { href: "/pricing", label: "Pricing" },
  { href: "/blog", label: "Blog" },
  { href: "/contact", label: "Contact" },
];

<Navbar navItems={navItems} />;
```
