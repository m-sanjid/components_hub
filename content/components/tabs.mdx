## Usage

```tsx showLineNumbers
import { Tabs } from "@/components/ui/tabs";
```

```tsx showLineNumbers
const tabs = [
  {
    value: "tab1",
    label: "Tab 1",
    content: <p>This is content for Tab 1.</p>,
  },
  {
    value: "tab2",
    label: "Tab 2",
    content: <p>This is content for Tab 2.</p>,
  },
  {
    value: "tab3",
    label: "Tab 3",
    content: <p>This is content for Tab 3.</p>,
  },
];

<Tabs tabs={tabs} />
```
