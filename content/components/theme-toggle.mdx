<Step>
  Wrap your app with **ThemeProvider** from **next-themes** inside
  **layout.tsx**
</Step>

```tsx
"use client";

import { ThemeProvider } from "next-themes";

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <ThemeProvider attribute="class" defaultTheme="system" enableSystem>
      {children}
    </ThemeProvider>
  );
}
```

## Usage

```tsx
"use client";

import ThemeToggle from "@/components/ui/theme-toggle";

export default function Page() {
  return (
    <div className="flex min-h-screen items-center justify-center bg-neutral-100 dark:bg-neutral-900">
      <ThemeToggle />
    </div>
  );
}
```

## Features

- Works with next-themes provider
- Framer Motion animations for smooth transitions
- Icons morph, rotate, and fade between states
- Full light/dark mode support
- Accessible with aria-label
